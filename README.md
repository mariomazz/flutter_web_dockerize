commans for docker operations :

    create image :

        - docker build -t flutter_web_dockerize .

    create container and run : 

        - docker run -d -p 8080:80 --name flutter_web_dockerize flutter_web_dockerize

    stop container : 

        - docker stop flutter_web_dockerize
